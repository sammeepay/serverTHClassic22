valkyrie,53,47,3	script	Scholar#Valkyrie	743,{
	if (ADVJOB == 0 || Upper != 1) {
		set .@karma_d,rand(1,10);
		if (.@karma_d > 4) {
			mes "[Scholar]";
			mes "ขอแสดงความยินดีด้วย";
			mes "เกียรติยศแด่เหล่านักรบ!";
			close;
		}
		mes "[Scholar]";
		mes "มันต้องใช้เวลาทั้งชีวิต...";
		mes "ทั้งชีวิตจริงๆ";
		mes "ในการสะสมความรู้";
		mes "ที่จำเป็นสำหรับการ";
		mes "จะเป็น Scholar...";
		next;
		mes "[Scholar]";
		mes "มันช่างล้นเหลือยิ่งนัก";
		mes "ยิ่งเจ้าเรียนรู้มากเท่าไร เจ้า";
		mes "ก็จะยิ่งค้นพบว่ามีสิ่งอื่นที่";
		mes "เจ้ายังไม่รู้อีกมากเพียงใด";
		mes "กระบวนการแห่งการเรียนรู้";
		mes "นั้นไม่มีวันสิ้นสุด...";
		close;
	}
	else if (ADVJOB == Job_Professor && Class == Job_Mage_High && JobLevel > 39) {
		mes "[Scholar]";
		mes "Midgard ยังมี";
		mes "Scholar ไม่เพียงพอที่จะ";
		mes "ช่วยนำพาไปสู่ยุคสมัยใหม่";
		mes "แห่งความรุ่งเรือง";
		mes "โลกกำลังต้องการเจ้า...";
		next;
		mes "[Scholar]";
		mes "เจ้าจะรับผิดชอบ";
		mes "หน้าที่อันยิ่งใหญ่นี้ไหม?";
		mes "เจ้าจะรับใช้ Midgard";
		mes "ในฐานะ Scholar หรือไม่?";
		next;
		if (select("ไม่:ใช่") == 1) {
			mes "[Scholar]";
			mes "เมื่อเจ้าพร้อม";
			mes "เชิญกลับมาได้ทุกเมื่อ";
			mes "เกียรติยศแด่เหล่านักรบ!";
			close;
		}
		if (SkillPoint) {
			mes "[Scholar]";
			mes "มันยังคงเป็นไปได้ที่คุณจะเรียนรู้ทักษะเพิ่มเติม โปรดใช้";
			mes "แต้มทักษะที่เหลืออยู่ทั้งหมดของคุณให้หมดก่อนที่จะกลับมาหาข้า";
			close;
		}
		jobchange Job_Professor;
		set ADVJOB,0;
		mes "[Scholar]";
		mes "ขอแสดงความยินดีด้วย!";
		mes "ในฐานะ Professor ข้าหวัง";
		mes "ว่าเจ้าจะมีบทบาทสำคัญ";
		mes "ในการนำแสงสว่างแห่ง";
		mes "ความรู้ไปสู่ที่ที่ยังคง";
		mes "มีความมืดมิดอยู่";
		close;
	}
	mes "[Scholar]";
	mes "ยินดีต้อนรับ";
	mes "สู่ Valhalla";
	mes "หอแห่งเกียรติยศ";
	next;
	mes "[Scholar]";
	mes "โปรดทำตัว";
	mes "ตามสบาย";
	mes "ในขณะที่เจ้าอยู่ที่นี่";
	mes "เกียรติยศแด่เหล่านักรบ!";
	close;
}
