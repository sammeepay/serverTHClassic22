function	script	F_IzludeChannel	{
	mes "[เจ้าหน้าที่บริการแชนแนล]";
	mes "เมือง Izlude แห่งนี้ช่างดูคึกคักมีชีวิตชีวามากเลยว่าไหมครับ?";
	mes "ผมมาที่นี่เพื่อช่วยให้การเดินทางในเมืองนี้ราบรื่นยิ่งขึ้น!";
	mes "ผมสามารถส่งคุณไปยังเมือง Izlude ในแชนแนลอื่นได้นะ";
	mes "คุณต้องการจะย้ายไปยังแชนแนลอื่นไหมครับ?";
	next;
	set .@i, select(getarg(0)+":ไม่เป็นไร ขอบคุณ");
	if (.@i < 6) {
		setarray .@maps$[1],"izlude","izlude_a","izlude_b","izlude_c","izlude_d";
		// ตั้งจุดเซฟและวาร์ปไปยังพิกัดกึ่งกลางเมือง
		savepoint .@maps$[.@i],128,98,1,1;
		warp .@maps$[.@i],128,98;
		end;
	}
	close;
}
