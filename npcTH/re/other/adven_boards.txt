function	script	F_AvenBoard	{
	.@mo_board = select( "ดูรายชื่อมอนสเตอร์แบ่งตามชั้น",
					"ระบบสืบค้นข้อมูลมอนสเตอร์",
					( countitem(909) == 30000 ? "ส่งคำร้องขอ" : "" ),
					"ยกเลิก" );
	if (.@mo_board == 1)
		return 1;
	else if (.@mo_board == 2) {
		while(1) {
			mes "[ ระบบสืบค้นข้อมูลมอนสเตอร์ ]";
			mes "ยินดีต้อนรับสู่ระบบสืบค้นข้อมูลมอนสเตอร์ ท่านสามารถค้นหาข้อมูลมอนสเตอร์ที่ปรากฏในดันเจี้ยนแห่งนี้ได้ กรุณากรอกชื่อมอนสเตอร์ที่ต้องการค้นหา (ตรวจสอบการเว้นวรรคให้ถูกต้อง)";
			next;
			input .@inputstr$;
			setarray .@element$[0],"ไร้ธาตุ","น้ำ","ดิน","ไฟ","ลม","พิษ","ศักดิ์สิทธิ์","เงา","วิญญาณ","ผีดิบ";
			setarray .@size$[0],"ขนาดกลาง","ขนาดเล็ก","ขนาดใหญ่";
			setarray .@race$[0],"ไร้รูปร่าง","ผีดิบ","สัตว์","พืช","แมลง","ปลา","ปีศาจ","กึ่งมนุษย์","เทวทูต","มังกร";
			.@size = getargcount();
			.@i = 0;
			while( .@i < .@size && getarg(.@i+1,"") != .@inputstr$ ) .@i += 3;
			mes "[ ระบบสืบค้นข้อมูลมอนสเตอร์ ]";
			mes "กำลังดำเนินการค้นหา...";
			next;
			mes "[ ระบบสืบค้นข้อมูลมอนสเตอร์ ]";
			mes "การค้นหาเสร็จสมบูรณ์";
			mes "กำลังแสดงผลการค้นหาข้อมูล";
			next;
			mes "[ ระบบสืบค้นข้อมูลมอนสเตอร์ ]";
			if (.@i >= .@size) {
				mes "ไม่พบข้อมูลของมอนสเตอร์ที่ระบุ";
				mes "กรุณาตรวจสอบชื่อมอนสเตอร์ใหม่อีกครั้ง";
			}
			else {
				.@mob_id = getarg(.@i);
				mes "- ชื่อ : "+ getarg(.@i+1);
				mes "- ธาตุ : "+ .@element$[ getmonsterinfo(.@mob_id,MOB_ELEMENT) % 10 ]; // ใช้ % 10 เพื่อให้ตรงกับ Array
				mes "- ขนาด : "+ .@size$[ getmonsterinfo(.@mob_id,MOB_SIZE) ];
				mes "- เผ่า : "+ .@race$[ getmonsterinfo(.@mob_id,MOB_RACE) ];
				mes "--------------------------------^FD8D02";
				mes getarg(.@i+2);
			}
			next;
			mes "[ ระบบสืบค้นข้อมูลมอนสเตอร์ ]";
			mes "ท่านต้องการค้นหาข้อมูลมอนสเตอร์ตัวอื่นต่อหรือไม่?";
			next;
			if (select( "ค้นหาต่อ","ยกเลิก" ) == 1)
				continue;
			mes "[ ระบบสืบค้นข้อมูลมอนสเตอร์ ]";
			mes "ระบบสืบค้นข้อมูลมอนสเตอร์กำลังจะปิดการทำงาน";
			next;
			return 2;
		}
	}
	else if (.@mo_board == 3) {
		mes "ข้อมูลที่ร้องขอกำลังจะแสดงผล - หากข้อความนี้ปรากฏขึ้นแสดงว่าการทำงานปกติ";
		next;
		return 3;
	}
	else if (.@mo_board == 4)
		close;
	else {
		mes "เกิดข้อผิดพลาดในการทำงาน (Operation Error)";
		close;
	}
}

}
