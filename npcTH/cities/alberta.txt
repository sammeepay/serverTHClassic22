		mes "แล้วค่อยมาหาข้าใหม่";
		close;
	}
	if (@event_zelopy == 0) {
		mes "มาทำบ้าอะไรที่นี่วะ?";
		mes "บนเรือลำนี้ไม่มีอะไรให้ฟรีๆ หรอกนะ ถ้าอยากได้อะไร ก็ต้องทำงานแลกมา!!";
		next;
		mes "[Phelix]";
		mes "หืม... เอาแบบนี้เป็นไง เจ้าเอา Jellopy มาให้ข้า 10 อัน แล้วข้าจะให้ Red Potion 1 ขวด ตกลงไหม?";
		mes "หรือถ้ามันยากเกินไปสำหรับมือใหม่อ่อนหัดอย่างเจ้า... ก็เอา Jellopy 3 อัน มาแลก Carrot 1 หัวละกัน";
		next;
		mes "[Phelix]";
		mes "ถ้าสนใจข้อเสนอของข้า ก็ไปหาของที่ข้าบอกมาซะ";

		close;
	}
	mes "หืม.. จะเอา Jellopy มาแลก Red Potion หรือ Carrot ดีล่ะ? ว่ามา... จะเอาอันไหน?";

	switch(select("ขอ Red Potion ครับ:ขอ Carrot ครับ")) {
	case 1:
		mes "[Phelix]";
		mes "ได้...";
		mes "ไหนดูซิว่า";

		mes "[Phelix]";

			mes "เฮ้ย! เมื่อกี้ไม่ได้ฟังหรือไง? ข้าบอกว่า Jellopy 10 อัน แลก Red Potion 1 ขวด.. หูหนวกเรอะ?";

		} else {
			.@max = countitem(909)/10;
			mes "อืม... ไม่เลวนี่...";
			mes "แล้วเจ้าอยากได้";

			next;
			switch(select("เท่าที่แลกได้ทั้งหมดเลยครับ:ข้ากำหนดจำนวนเอง:ไม่เอาละ ข้าชอบ Jellopy ของข้ามากกว่า")) {

				delitem 909,.@max*10;// Jellopy
				getitem 501,.@max;// Red_Potion

				input .@amount;
				next;
				mes "[Phelix]";
				if (.@amount <= 0) {
					mes "ขอบใจมาก ไว้แวะมาใหม่ละกัน";
					close;
				}
				if (.@amount > 100) {
					mes "เฮ้ย ข้าบอกว่ายังไง? ครั้งละไม่เกิน 100 อันไง เจ้ากะจะฆ่าข้าหรือไงกัน!";
					close;
				}
				if (countitem(909) < .@amount*10) {
					mes "หืม ดูเหมือนเจ้าจะมีไม่พอนะ ไปหา Jellopy มาเพิ่มซะถ้ายังอยากได้ของจากข้า";
					close;
				}
				delitem 909,.@amount*10;// Jellopy
				getitem 501,.@amount;// Red_Potion
				break;
			case 3:
				mes "[Phelix]";
				mes "ไม่มีปัญหา,";
				mes "ไว้เจอกันคราวหน้า";
				close;
			}
			mes "[Phelix]";
			mes "เอ้า นี่ของเจ้า! ตามที่สัญญาไว้ อย่าดื่มทีเดียวจนหมดขวดล่ะ";

		mes "[Phelix]";
		mes "เอาล่ะ ไหนดูซิว่าเจ้ามีอะไรมาบ้าง...";
		next;
		mes "[Phelix]";
		if (countitem(909) < 3) {
			mes "ฟังนะเจ้าหนูหัดเดิน ข้าบอกว่า Jellopy 3 อัน แลก Carrot 1 หัว.. เข้าใจไหม?";
			close;
		}
		.@max = countitem(909)/3;
		mes "ไม่เลวนี่เจ้าอ่อน...";
		mes "จะเอาเท่าไหร่ล่ะ?";
		next;
		switch(select("เท่าที่แลกได้ทั้งหมดเลยครับ:ข้ากำหนดจำนวนเอง:ไม่เอาละ ข้าชอบ Jellopy ของข้ามากกว่า")) {
		case 1:
			delitem 909,.@max*3;// Jellopy
			getitem 515,.@max;// Carrot
			break;
		case 2:
			mes "[Phelix]";
			mes "ฟังนะ ข้าไม่ให้เจ้าเกินครั้งละ 100 อันหรอกนะ เข้าใจไหม? ถ้าไม่อยากได้แล้วก็ใส่เลข '0'";
			input .@amount;
			next;
			mes "[Phelix]";
			if (.@amount == 0) {
				mes "งั้นก็ได้ ไว้เจอกันคราวหน้า";
				close;
			}
			if (.@amount > 100) {
				mes "เฮ้ยเจ้าอ่อน ข้าบอกว่ามากสุด 100 อันไง ไม่เกินนี้! ข้าไม่ยอมทำงานจนหลังหักเพื่อคนอย่างเจ้าหรอกนะ!";
				close;
			}
			if (countitem(909) < .@amount*3) {
				mes "ดูเหมือนเจ้าจะมีไม่พอนะ ไปหามาเพิ่มซะถ้าอยากได้อย่างอื่น";
				close;
			}
			delitem 909,.@amount*3;// Jellopy
			getitem 515,.@amount;// Carrot
			break;
		case 3:
			mes "[Phelix]";
			mes "ไว้เจอกัน";
			close;
		}
		mes "[Phelix]";
		mes "เอ้า เอาไป~! ตามที่สัญญาไว้ อย่ากินจนติดคอล่ะ";
		close;
	}
