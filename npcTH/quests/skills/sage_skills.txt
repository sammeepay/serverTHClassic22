			}
			else {
				mes "[Mishuna]";
				mes "ตกลง ถ้าอย่างนั้น โปรด";
				mes "นำไอเทมต่อไปนี้มาให้ข้า";
				mes "เพื่อที่เจ้าจะได้เรียนรู้ทักษะ";
				mes .@Skill$[.@i]+" Elemental Change";
				next;
				mes "[Mishuna]";

				mes "^FF0000Payon Solution 1 อัน^000000 และ";
				mes "^FF0000Morocc Solution 1 อัน^000000 โปรด";
				mes "กลับมาหาข้าอีกครั้งเมื่อเจ้า";

			}
		}

			mes "เจ้ากำลังมีปัญหาในการ";

			mes "เผื่อเอาไว้ ข้าจะขอย้ำอีกครั้งว่า";
			mes "เจ้าต้องนำอะไรมาให้ข้าบ้างเพื่อ";
			mes "เรียนทักษะ "+.@Skill$[.@i]+" Elemental Change";
			next;
			mes "[Mishuna]";
			mes "^ff0000"+.@ReqItem$[.@i]+" 20 อัน^000000,";
			mes "^ff0000Payon Solution 1 อัน^000000 และ";
			mes "^ff0000Morocc Solution 1 อัน^000000 โปรด";
			mes "อย่าลืมและเตรียมวัตถุดิบให้";
			mes "พร้อมในครั้งต่อไปที่เจ้ามาพบข้า";
			mes "ตกลงไหม?";
			close;
		}
		mes "[Mishuna]";
		mes "อา เจ้ากลับมาแล้ว ในที่สุด";
		mes "ข้าก็สามารถสอนเจ้าเกี่ยวกับทักษะ";
		mes .@Skill$[.@i]+" Elemental Change ได้เสียที";
		mes "ทักษะนี้มีโอกาสที่จะเปลี่ยนธาตุ";
		mes "ของมอนสเตอร์เป้าหมายให้กลาย";
		mes "เป็นธาตุ "+.@Skill$[.@i]+" อย่างถาวร";
		next;
		mes "[Mishuna]";
		mes "จำไว้ว่าเจ้าต้องใช้ Elemental";
		mes "Converter ในการร่ายทักษะนี้";
		mes "และมันมีอัตราความสำเร็จ เช่นเดียวกับ";
		mes "ทักษะ Elemental Converter Creation";
		mes "ดังนั้นจงระวังเรื่องนั้นด้วย";
		next;
		mes "[Mishuna]";
		mes "เอาล่ะ "+strcharinfo(0)+",";
		mes "ข้ากำลังจะร่ายมนตร์ที่จะช่วย";
		mes "ให้เจ้าจดจำทักษะ "+.@Skill$[.@i]+" Elemental";
		mes "Change ได้ พยายามอยู่นิ่งๆ นะ...";
		next;
		mes "^3355FFMishuna เริ่มร่ายเวทมนตร์";
		mes "ที่ฟังดูแปลกประหลาด ขณะที่";
		mes "แสงสีฟ้าอ่อนๆ เริ่มโอบล้อมรอบ";
		mes "ร่างกายของเขา และค่อยๆ สว่าง";
		mes "เจิดจ้าและเข้มข้นขึ้นเรื่อยๆ^000000";
		specialeffect2 EF_RUWACH;
		next;
		mes "[Mishuna]";
		mes "@#$%^~ Yap!";
		specialeffect2 EF_BRANDISHSPEAR;
		delitem .@ReqItem[.@i],20;
		delitem 1089,1; // Payon Potion
		delitem 1088,1; // Morocc Potion
		set SAG_SK,100;
		skill .@Skill[.@i],1,0;
		next;
		mes "[Mishuna]";
		mes strcharinfo(0)+"...";
		mes "ข้ายินดีที่จะบอกว่าเจ้าได้";
		mes "จดจำทักษะ "+.@Skill$[.@i]+" Elemental";
		mes "Change ได้สำเร็จแล้ว";
		mes "ข้าหวังว่ามันจะเป็นประโยชน์ต่อเจ้า";
